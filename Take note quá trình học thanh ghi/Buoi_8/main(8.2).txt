// Định nghĩa thanh ghi cấp Clock thủ công (để sửa lỗi Warning)
#define RCC_APB2ENR  (*((volatile unsigned long *)0x40021018))
struct
{
  unsigned char T ;
  unsigned  char  PWMValue;
  unsigned  char Counter;
  
}LedPWM = {250, 50 , 0 };
void SysTick_Handler()  //1ms
{
  LedPWM .Counter++;
  if (LedPWM .Counter == LedPWM .PWMValue)
    GPIOB.ODR.BITS.b12 = 0;
  if (LedPWM .Counter >= LedPWM.T)
  {
    GPIOB.ODR.BITS.b12 = 1;
    LedPWM.Counter = 0;
  }
}
