unsigned char Index  = 0;
unsigned long ServoTime[5] = {8800,8800,8800,8000, 124800, };
unsigned char un_used;
unsigned int* Pinconfig[5] = {BITBAND_PERIPHERAL_POINTER(&GPIOB.ODR.REG, 0),
                                BITBAND_PERIPHERAL_POINTER(&GPIOB.ODR.REG, 1),
                                BITBAND_PERIPHERAL_POINTER(&GPIOB.ODR.REG, 10),
                                BITBAND_PERIPHERAL_POINTER(&GPIOB.ODR.REG, 11),
                                0 };


//20ms=160000
void SysTick_Handler()  //1ms
{
  STK.LOAD = ServoTime[Index];
  if(Index)
    Pinconfig[Index -1] [0] = 0;
  Pinconfig[Index][0] = 1;
  Index++;
  if(Index == 5)
    Index =0;
  
  STK.CTRL.REG = BIT2 | BIT1;
  STK.VAL = 0;
  STK.CTRL.REG = BIT2 | BIT1 | BIT0;
}
void main() 
{

  STK.LOAD = 8000;
  STK.VAL = 8000;
  STK.CTRL.REG = BIT2 | BIT1 | BIT0;

  while(1)
  {
    
  }
}
