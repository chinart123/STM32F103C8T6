struct
{
  unsigned  char  PWMValue; //3bit lơw is used 
  unsigned  char Index;
  
}LedPWM = {.PWMValue = 0x05, .Index = 0};
void SysTick_Handler()  //1ms
{
  if(LedPWM.Index==0)
  {
	GPIOB.ODR.BITS.b12 = LedPWM.PWMValue & BIT0;
    STK.LOAD = 8000;       //1ms
    LedPWM.Index =1;
  }
  else if(LedPWM.Index==1)
  {
    GPIOB.ODR.BITS.b12 = LedPWM.PWMValue & BIT1;
    STK.LOAD = 16000;       //1ms
    LedPWM.Index =1;
  }
  else if(LedPWM.Index==2)
  {
    GPIOB.ODR.BITS.b12 = LedPWM.PWMValue & BIT2;
    STK.LOAD = 32000;       //1ms
    LedPWM.Index =0;
  }
  STK.CTRL.REG = BIT2 | BIT1;
  STK.VAL = 0;
  STK.CTRL.REG = BIT2 | BIT1 | BIT0;
  
}
void main() 
{
  GPIOB_Enable();
  GPIO_Mode(&GPIOB, BIT12, GPIO_MODE_OUTPUT_PUSHPULL_50MHz);
  STK.LOAD = 8000;
  STK.VAL = 8000;
  STK.CTRL.REG = BIT2 | BIT1 | BIT0;

  while(1)
  {
    
  }
}
