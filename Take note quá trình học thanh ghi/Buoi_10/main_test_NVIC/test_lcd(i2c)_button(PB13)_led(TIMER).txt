#include <stm32f103c8t6.h>
/*test tổng hợp*/
// ==========================================================
// CẤU HÌNH & ĐỊNH NGHĨA
// ==========================================================
#define I2C_ADDR        0x27  // Địa chỉ LCD (0x27 hoặc 0x3F)
#define BUFFER_SIZE     128   

#define LCD_BACKLIGHT   0x08
#define LCD_EN          0x04
#define LCD_RS          0x01

// Biến quản lý Buffer I2C
volatile unsigned char TxBuffer[BUFFER_SIZE];
volatile int TxIndex = 0;   
volatile int TxLength = 0;  
volatile int I2C_State = 0; // 0: Idle, 1: Busy

// Biến quản lý nút nhấn
int press_count = 0; // Đếm số lần nhấn

// ==========================================================
// HÀM DELAY & NVIC
// ==========================================================
void Delay_ms(unsigned long ms) {
    for(volatile unsigned long i = 0; i < ms; i++) {
        for(volatile unsigned long j = 0; j < 500; j++); 
    }
}

void NVIC_Config() {
    *((volatile unsigned long *)0xE000E100) = (1 << 31); // I2C1 EV
    *((volatile unsigned long *)0xE000E104) = (1 << 0);  // I2C1 ER
}

// ==========================================================
// I2C INTERRUPT SERVICE ROUTINES
// ==========================================================
void I2C1_EV_IRQHandler() {
    volatile unsigned int sr1 = I2C1.SR1.REG;
    volatile unsigned int sr2 = I2C1.SR2.REG; 

    if (sr1 & 1) { // SB
        I2C1.DR = (I2C_ADDR << 1); 
    }
    else if (sr1 & 2) { // ADDR
        TxIndex = 0;
        if (TxLength > 0) I2C1.DR = TxBuffer[TxIndex++];
        else I2C1.CR1.REG |= (1 << 9); 
    }
    else if (sr1 & (1 << 2)) { // BTF
        if (TxIndex < TxLength) {
            I2C1.DR = TxBuffer[TxIndex++]; 
        } else {
            I2C1.CR1.REG |= (1 << 9); // STOP
            I2C1.CR2.REG &= ~((1 << 9) | (1 << 10)); 
            I2C_State = 0; 
        }
    }
}

void I2C1_ER_IRQHandler() {
    if (I2C1.SR1.REG & (1 << 10)) { 
        I2C1.SR1.REG &= ~(1 << 10); 
        I2C1.CR1.REG |= (1 << 9);   
        I2C_State = 0;
    }
}

// ==========================================================
// KHỞI TẠO PHẦN CỨNG (GPIO, I2C, TIMER)
// ==========================================================
void System_Init() {
    // 1. Cấp Clock cho GPIOA, GPIOB, AFIO, I2C1, TIM2
    RCC.APB2_ENR.REG |= (1 << 2) | (1 << 3) | (1 << 0); // GPIOA, GPIOB, AFIO
    RCC.APB1_ENR.REG |= (1 << 21) | (1 << 0);           // I2C1, TIM2

    // 2. Cấu hình I2C (PB6, PB7)
    GPIOB.CRL.REG &= ~(0xFF << 24); 
    GPIOB.CRL.REG |=  (0xFF << 24); // AF Open-Drain

    I2C1.CR2.REG  = 8;      
    I2C1.CCR.REG  = 40;     
    I2C1.TRISE = 9;     
    I2C1.CR1.REG |= 1;  
    
    // 3. Cấu hình Nút nhấn PB13 (Input Pull-up)
    // CRH Bit 20-23. Mode=00(Input), CNF=10(Pull-up/down) -> 0x8
    GPIOB.CRH.REG &= ~(0xF << 20);
    GPIOB.CRH.REG |=  (0x8 << 20);
    GPIOB.ODR.REG |=  (1 << 13); // Ghi 1 vào ODR để kích hoạt Pull-UP
    
    // 4. Cấu hình ban đầu cho PA0 (Tắt, Push-Pull thường)
    GPIOA.CRL.REG &= ~(0xF << 0);
    GPIOA.CRL.REG |=  (0x3 << 0); // Output Push-Pull 50MHz
    GPIOA.ODR.REG &= ~(1 << 0);   // Tắt LED

    NVIC_Config();
}

// ==========================================================
// CÁC HÀM HỖ TRỢ LCD (BUFFER)
// ==========================================================
void I2C_FlushBuffer() {
    while (I2C_State == 1); 
    while (I2C1.SR2.REG & 2); 
    I2C_State = 1; 
    I2C1.CR2.REG |= (1 << 9) | (1 << 10); 
    I2C1.CR1.REG |= (1 << 8); 
}

void Buffer_Push(unsigned char data) {
    if (TxLength < BUFFER_SIZE) TxBuffer[TxLength++] = data;
}

void LCD_Send_To_Buffer(unsigned char val, unsigned char mode) {
    unsigned char params = LCD_BACKLIGHT | (mode ? LCD_RS : 0);
    Buffer_Push((val & 0xF0) | params | LCD_EN);
    Buffer_Push((val & 0xF0) | params & ~LCD_EN);
    Buffer_Push(((val << 4) & 0xF0) | params | LCD_EN);
    Buffer_Push(((val << 4) & 0xF0) | params & ~LCD_EN);
}

void LCD_Command(unsigned char cmd) {
    TxLength = 0; 
    LCD_Send_To_Buffer(cmd, 0);
    I2C_FlushBuffer(); 
    Delay_ms(2); 
}

void LCD_Print(char *str) {
    // Clear buffer trước khi in chuỗi mới
    TxLength = 0;
    while (*str) {
        LCD_Send_To_Buffer(*str++, 1);
    }
    I2C_FlushBuffer();
    Delay_ms(2);
}

void LCD_Init() {
    Delay_ms(50);
    TxLength = 0;
    Buffer_Push(0x30 | LCD_BACKLIGHT | LCD_EN); Buffer_Push(0x30 | LCD_BACKLIGHT);
    Buffer_Push(0x30 | LCD_BACKLIGHT | LCD_EN); Buffer_Push(0x30 | LCD_BACKLIGHT);
    Buffer_Push(0x30 | LCD_BACKLIGHT | LCD_EN); Buffer_Push(0x30 | LCD_BACKLIGHT);
    Buffer_Push(0x20 | LCD_BACKLIGHT | LCD_EN); Buffer_Push(0x20 | LCD_BACKLIGHT);
    I2C_FlushBuffer();
    Delay_ms(10);
    LCD_Command(0x28); 
    LCD_Command(0x0C); 
    LCD_Command(0x01); 
    Delay_ms(5); 
}

// ==========================================================
// CÁC HÀM CẤU HÌNH ĐẶC BIỆT CHO PA0 (Switch Mode)
// ==========================================================

// Chế độ 1: GPIO thường (để nháy LED 0.5s)
void PA0_Set_GPIO_Mode() {
    // Tắt Timer 2 nếu đang chạy
    TIM2.CR1.BITS.CEN = 0;
    
    // Cấu hình PA0 là Output Push-Pull
    GPIOA.CRL.REG &= ~(0xF << 0);
    GPIOA.CRL.REG |=  (0x3 << 0);
}

// Chế độ 2: PWM Mode (để LED thở)
void PA0_Set_PWM_Mode() {
    // Cấu hình PA0 là Alternate Function Push-Pull (AF_PP)
    // Mode=11, CNF=10 -> 0xB
    GPIOA.CRL.REG &= ~(0xF << 0);
    GPIOA.CRL.REG |=  (0xB << 0);

    // Setup TIM2 PWM (Channel 1)
    // TIM2 nằm ở APB1, Clock = 8MHz (Mặc định)
    TIM2.PSC.REG = 7;      // Chia 8 -> 1MHz (1us)
    TIM2.ARR.REG = 999;    // 1000 tick -> 1ms (1kHz)
    
    // --- SỬA LỖI TẠI ĐÂY ---
    // Dùng đúng cấu trúc trong general_timer.h: BITS_OUTPUT
    TIM2.CCMR1.BITS_OUTPUT.OC1M = 6;  // PWM Mode 1 (110)
    TIM2.CCMR1.BITS_OUTPUT.OC1PE = 1; // Preload Enable
    // -----------------------

    TIM2.CCER.BITS.CC1E = 1;        // Enable Output CH1
    TIM2.CR1.BITS.CEN = 1;          // Start Timer
}

// Hàm kiểm tra nút nhấn (Non-blocking)
// Trả về 1 nếu có nhấn mới, 0 nếu không
int Check_Button() {
    // Đọc PB13 (Input Pull-up -> Nhấn = 0)
    if ((GPIOB.IDR.REG & (1 << 13)) == 0) {
        Delay_ms(20); // Chống rung
        if ((GPIOB.IDR.REG & (1 << 13)) == 0) {
            // Đợi nhả nút
            while ((GPIOB.IDR.REG & (1 << 13)) == 0);
            return 1;
        }
    }
    return 0;
}

// ==========================================================
// MAIN FUNCTION
// ==========================================================
void main() {
    System_Init();
    LCD_Init();

    LCD_Print("READY...");
    
    while(1) {
        // Kiểm tra nút nhấn ở vòng lặp chính
        if (Check_Button()) {
            press_count++;
            
            // Nếu quá 6 thì quay về 1
            if (press_count > 6) press_count = 1;

            // Xử lý sự kiện theo số lần nhấn
            switch (press_count) {
                case 1:
                    PA0_Set_GPIO_Mode(); // Đảm bảo tắt PWM
                    GPIOA.ODR.REG &= ~(1 << 0); // Tắt LED
                    LCD_Command(0x01); // Xóa màn hình
                    LCD_Print("1st PRESS");
                    break;
                case 2:
                    LCD_Command(0x01);
                    LCD_Print("2nd PRESS");
                    break;
                case 3:
                    LCD_Command(0x01);
                    LCD_Print("3rd PRESS");
                    break;
                case 4:
                    LCD_Command(0x01);
                    LCD_Print("4th PRESS");
                    break;
                case 5:
                    LCD_Command(0x01);
                    LCD_Print("5th PRESS");
                    LCD_Command(0xC0);
                    LCD_Print("Blinking LED...");
                    // Chuyển sang xử lý loop bên dưới
                    break;
                case 6:
                    LCD_Command(0x01);
                    LCD_Print("6th PRESS");
                    LCD_Command(0xC0);
                    LCD_Print("Breathing LED...");
                    PA0_Set_PWM_Mode(); // Bật PWM
                    break;
            }
        }

        // --- XỬ LÝ TRẠNG THÁI LIÊN TỤC (BLINK / BREATHE) ---
        
        // TRẠNG THÁI 5: NHÁY LED 0.5s
        if (press_count == 5) {
            PA0_Set_GPIO_Mode();
            GPIOA.ODR.REG ^= (1 << 0); // Đảo trạng thái LED
            
            // Delay 500ms nhưng vẫn phải check nút nhấn
            // Chia nhỏ delay ra để check nút nhạy hơn
            for (int i=0; i<50; i++) { 
                Delay_ms(10); 
                if (Check_Button()) { 
                    press_count = 6; // Chuyển sang trạng thái 6 ngay
                    goto CASE_6_START; // Nhảy cóc (Goto) để xử lý ngay
                }
            }
        }

        // TRẠNG THÁI 6: LED THỞ (PWM)
        else if (press_count == 6) {
            CASE_6_START: // Nhãn dán để nhảy tới
            // Nếu chưa bật PWM thì bật lại (cho chắc)
            if (TIM2.CR1.BITS.CEN == 0) PA0_Set_PWM_Mode();

            // Hiệu ứng thở
            int brightness = 0;
            int step = 2;
            
            // Vòng lặp thở (làm sao để thoát? Check button bên trong)
            while (press_count == 6) {
                TIM2.CCR1.REG = brightness;
                brightness += step;
                if (brightness >= 1000 || brightness <= 0) step = -step;
                
                // Delay thở (theo yêu cầu 10000)
                for(volatile long i=0; i<5000; i++); 

                // Check nút nhấn ngay trong lúc thở
                if (Check_Button()) {
                    press_count = 1; // Quay về 1
                    break; // Thoát vòng lặp thở
                }
            }
        }
    }
}
