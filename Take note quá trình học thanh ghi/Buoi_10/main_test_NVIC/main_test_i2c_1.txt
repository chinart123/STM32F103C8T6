#include <stm32f103c8t6.h>
/*@@

// --- BIẾN TOÀN CỤC ---
// Không define cứng nữa, ta sẽ dùng biến để lưu địa chỉ tìm được
unsigned char I2C_ADDR = 0x27; // Giá trị mặc định, sẽ được cập nhật tự động

// Các bit điều khiển PCF8574
#define LCD_BACKLIGHT   0x08
#define LCD_EN          0x04
#define LCD_RW          0x02
#define LCD_RS          0x01

// ==========================================================
// HÀM DELAY
// ==========================================================
void Delay_ms(unsigned long ms) {
    for(unsigned long i = 0; i < ms; i++) {
        for(unsigned long j = 0; j < 500; j++); 
    }
}

// ==========================================================
// PHẦN 1: GIAO TIẾP I2C (Low Level)
// ==========================================================
void I2C1_Init() {
    // 1. Cấp Clock
    RCC.APB2_ENR.REG |= (1 << 3);  // GPIOB Enable
    RCC.APB1_ENR.REG |= (1 << 21); // I2C1 Enable

    // 2. Cấu hình PB6, PB7 (AF Open-Drain)
    GPIOB.CRL.REG &= ~(0xFF << 24); 
    GPIOB.CRL.REG |=  (0xFF << 24); 

    // 3. Cài đặt tốc độ (Standard 100kHz)
    // Lưu ý: Dùng trực tiếp không qua .REG (theo thư viện thầy bạn)
    I2C1.CR2.REG  = 8;      
    I2C1.CCR.REG  = 40;     
    I2C1.TRISE = 9;     

    // 4. Bật I2C
    I2C1.CR1.REG |= 1;  
}

void I2C1_Start() {
    I2C1.CR1.REG |= (1 << 8);    
    while (!(I2C1.SR1.REG & 1)); 
}

void I2C1_Stop() {
    I2C1.CR1.REG |= (1 << 9);    
}

void I2C1_WriteData(unsigned char data) {
    while (!(I2C1.SR1.REG & (1 << 7))); // Đợi TxE
    I2C1.DR = data;                     
    while (!(I2C1.SR1.REG & (1 << 2))); // Đợi BTF
}

// ==========================================================
// PHẦN 2: MÁY QUÉT ĐỊA CHỈ (SCANNER)
// ==========================================================
unsigned char I2C_Scanner() {
    // Quét thử các địa chỉ phổ biến trước cho nhanh
    unsigned char addr;
    
    for (addr = 1; addr < 127; addr++) {
        // 1. Gửi Start
        I2C1.CR1.REG |= (1 << 8); 
        while (!(I2C1.SR1.REG & 1)); 
        
        // 2. Gửi thử Địa chỉ + Bit Write (0)
        I2C1.DR = (addr << 1); 
        
        // 3. Chờ xem có ai trả lời không?
        // Ta kiểm tra 2 cờ: ADDR (Có người trả lời) hoặc AF (Thất bại)
        int timeout = 0;
        while (1) {
            // Nếu cờ ADDR (Bit 1) bật lên -> TÌM THẤY!
            if (I2C1.SR1.REG & 2) { 
                // Xóa cờ ADDR
                volatile int temp = I2C1.SR1.REG;
                temp = I2C1.SR2.REG;
                I2C1_Stop();
                return addr; // Trả về địa chỉ ngay lập tức
            }
            
            // Nếu cờ AF (Acknowledge Failure - Bit 10) bật lên -> KHÔNG CÓ AI
            if (I2C1.SR1.REG & (1 << 10)) { 
                I2C1.SR1.REG &= ~(1 << 10); // Xóa cờ lỗi
                I2C1_Stop();
                break; // Thoát vòng lặp chờ để thử số tiếp theo
            }
            
            timeout++;
            if (timeout > 2000) break; // Lâu quá không phản hồi thì bỏ qua
        }
    }
    return 0; // Không tìm thấy gì cả
}

// ==========================================================
// PHẦN 3: ĐIỀU KHIỂN LCD (Dùng địa chỉ đã tìm được)
// ==========================================================
void PCF8574_Send(unsigned char data) {
    I2C1_Start();
    // Dùng biến I2C_ADDR thay vì số cứng
    I2C1.DR = I2C_ADDR << 1;     
    while (!(I2C1.SR1.REG & 2)); // Đợi ADDR
    
    volatile int temp = I2C1.SR1.REG;
    temp = I2C1.SR2.REG;         
    
    I2C1_WriteData(data);
    I2C1_Stop();
}

void LCD_Pulse_Enable(unsigned char data) {
    PCF8574_Send(data | LCD_EN);  
    Delay_ms(1);                  
    PCF8574_Send(data & ~LCD_EN); 
    Delay_ms(1);                  
}

void LCD_Send(unsigned char val, unsigned char mode) {
    unsigned char high_nibble = val & 0xF0;
    unsigned char low_nibble  = (val << 4) & 0xF0;
    unsigned char params = LCD_BACKLIGHT | (mode ? LCD_RS : 0);

    LCD_Pulse_Enable(high_nibble | params);
    LCD_Pulse_Enable(low_nibble | params);
}

void LCD_Command(unsigned char cmd) {
    LCD_Send(cmd, 0);
}

void LCD_Data(unsigned char data) {
    LCD_Send(data, 1);
}

void LCD_Init() {
    Delay_ms(50); 
    LCD_Pulse_Enable(0x30 | LCD_BACKLIGHT); Delay_ms(5); 
    LCD_Pulse_Enable(0x30 | LCD_BACKLIGHT); Delay_ms(1); 
    LCD_Pulse_Enable(0x30 | LCD_BACKLIGHT); 
    LCD_Pulse_Enable(0x20 | LCD_BACKLIGHT); Delay_ms(10); 

    LCD_Command(0x28); 
    LCD_Command(0x0C); 
    LCD_Command(0x06); 
    LCD_Command(0x01); // Xóa màn hình
    Delay_ms(5);       
}

void LCD_Print(char *str) {
    while (*str) LCD_Data(*str++);
}

// Hàm phụ: In số Hex lên màn hình
void LCD_PrintHex(unsigned char num) {
    char hex[] = "0123456789ABCDEF";
    LCD_Data('0'); LCD_Data('x');
    LCD_Data(hex[(num >> 4) & 0xF]); 
    LCD_Data(hex[num & 0xF]); 
}

// ==========================================================
// CHƯƠNG TRÌNH CHÍNH
// ==========================================================
void main() {
    // 1. Khởi tạo phần cứng
    I2C1_Init();
    
    // 2. CHẠY MÁY QUÉT ĐỂ TÌM ĐỊA CHỈ
    unsigned char found_address = I2C_Scanner();
    
    if (found_address != 0) {
        // --> ĐÃ TÌM THẤY!
        I2C_ADDR = found_address; // Cập nhật biến toàn cục
        
        // Khởi tạo LCD với địa chỉ ĐÚNG vừa tìm được
        LCD_Init();
        
        // In kết quả ra màn hình
        LCD_Print("Scanner Found:");
        LCD_Command(0xC0); // Xuống dòng 2
        LCD_PrintHex(found_address); // Hiện ví dụ: 0x27 hoặc 0x3F
        
    } else {
        // --> KHÔNG TÌM THẤY GÌ CẢ
        // Có thể lỏng dây hoặc hỏng module
        // Bật LED PC13 (nếu có) để báo lỗi
        RCC.APB2_ENR.REG |= (1<<4);
        GPIOC.CRH.REG &= ~(0xF << 20);
        GPIOC.CRH.REG |= (2 << 20);
        while(1) {
            GPIOC.ODR.REG ^= (1<<13); // Nháy LED báo động
            Delay_ms(100);
        }
    }

    while(1) {
        // Chương trình đã xong, đứng chơi
    }
}
